💻 Ejemplos Prácticos

Ejemplo 1: LED Intermitente

void setup() {
  pinMode(13, OUTPUT);
}

void loop() {
  digitalWrite(13, HIGH);
  delay(1000);
  digitalWrite(13, LOW);
  delay(1000);
}

Ejemplo 2: Control de Servomotor

#include 

Servo motor;

void setup() {
  motor.attach(9);
}

void loop() {
  motor.write(0);
  delay(1000);
  motor.write(90);
  delay(1000);
  motor.write(180);
  delay(1000);
}
